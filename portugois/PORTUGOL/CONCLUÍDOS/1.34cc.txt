ALGORITMO
DECLARE x,i,elevado,denominador,cosseno,contador,fat NUMÉRICO
ESCREVA "digite um valor em graus para achar o seno"
LEIA x	
elevado<==2
denominador<==2
cosseno<==x
contador<==1
fat<==1
ENQUANTO (contador<=15) FAÇA:
	INÍCIO
	i <== denominador
	ENQUANTO (i>=1) FAÇA:
       	fat<==fat*i
       	i<==i-1
   	ESCREVA "Fatorial:", fat

	SE contador%2==0:
   	ENTÃO cosseno<==cosseno+(x**elevado)/(fat)
	SENÃO
	ENTÃO cosseno<==cosseno-(x**elevado)/(fat)

   	contador<==contador+1
   	elevado<==elevado+2
   	denominador<==denominador+2
	FIM
ESCREVA cosseno
FIM_ALGORITMO
