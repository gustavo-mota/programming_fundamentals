ALGORITMO

soma<--0
ESCREVA "Digite um número"
LEIA n
n<--n*3,14/180
i<--0

INÍCIO
	fat <-- 1
	PARA j<--1 ATÉ k FAÇA PASSO 1
	INÍCIO
		fat <-- fat * j
	FIM
	x[i]<--n**k/fat
	i<--i+1
FIM

	SE (l MOD 2 = 0)
