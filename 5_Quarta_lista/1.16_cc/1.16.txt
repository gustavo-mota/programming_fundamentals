DECLARE aluno[n] NUMÉRICO
PARA i <-- 1 ATÉ n FAÇA PASSO 1
INÍCIO
	PARA j <-- 1 ATÉ 30 FAÇA PASSO 1
	INÍCIO
		LEIA n2
		SE (n2=v[j-1])

	FIM
	aluno[i] <-- cont
	cont <-- 0
FIM
ESCREVA “RESULTADOS”
PARA i <-- 1 ATÉ n FAÇA PASSO 1

	ESCREVA “O ALUNO”, i, “ACERTOU”, aluno[i], “QUESTÕES”
