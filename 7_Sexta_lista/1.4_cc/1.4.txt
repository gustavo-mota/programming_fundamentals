SUB-ROTINA multiplica(numero NUMÉRICO, multi NUMÉRICO)
DECLARE resultado, i, vezes NUMÉRICO 
SE multi = 0
ENTÃO resultado<--0
SENÃO SE multi = 1
ENTÃO resultado <-- numero
SENÃO	INÍCIO
	vezes<--0
	ENQUANTO multi!=1 FAÇA
	INÍCIO
		multi<-- multi - 1
		vezes<-- vezes + numero
	FIM
	PARA i<--1 ATÉ vezes FAÇA
	INÍCIO
		numero <-- numero + 1
	FIM
	resultado <-- numero
	FIM
RETORNE resultado
FIM_SUB-ROTINA
