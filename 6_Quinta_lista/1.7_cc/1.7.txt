ALGORITMO
DECLARE matriz[4,4],i,j,s,r,x,y NUMÉRICO
PARA i <-- 1 ATÉ 4 FAÇA
INÍCIO
	PARA j <-- 1 ATÉ 4 FAÇA
	INÍCIO
		LEIA matriz[i,j]
	FIM
FIM
x <-- 4
y <-- 1
PARA s <-- 1 ATÉ 4 FAÇA
INÍCIO
	PARA r <-- 1 ATÉ 4 FAÇA
	INÍCIO
		SE s!=y E r!=x
		ENTÃO ESCREVA matriz[s,r]
	FIM
	x <-- x-1
	y <-- y+1
FIM
FIM_ALGORITMO
