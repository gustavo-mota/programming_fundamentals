ALGORITMO
DECLARE soma_linha[3],soma_coluna[3],soma_diagonal,soma_secundária,matriz[3,3],x,y,i,j,s,r,comparador[8],h,v NUMÉRICO
	quadrado LÓGICO
quadrado <-- VERDADEIRO
soma_linha <-- [0,0,0]
soma_coluna <-- [0,0,0]
soma_diagonal <-- 0
soma_secundária <-- 0
PARA i<--1 ATÉ 3 FAÇA
INÍCIO
	PARA j<--1 ATÉ 3 FAÇA
	INÍCIO
		LEIA matriz[i,j]
		soma_linha[i] <-- soma_linha[i] + matriz[i,j]
		SE i=j
		ENTÃO soma_diagonal <-- soma_diagonal + matriz[i,j]
FIM
x <-- 1
y <-- 3
PARA s<--1 ATÉ 3 FAÇA
INÍCIO
	PARA r<--1 ATÉ 3 FAÇA
	INÍCIO
		soma_coluna[s] <-- soma_coluna[s] + matriz[r,s]
		soma_secundária <-- soma_secundária + matriz[x,y]
		x <-- x + 1
		y <-- y - 1
	FIM
FIM
PARA d <-- 1 ATÉ 3 FAÇA
INÍCIO
	comparador[d] <-- soma_linha[d]
	comparador[d+3] <-- soma_coluna[d]
FIM
comparador[7] <-- soma_diagonal
comparador[8] <-- soma_secundária
PARA h <--- 1 ATÉ 8 FAÇA
INÍCIO
	PARA v <--1 ATÉ 8 FAÇA
	INÍCIO
		SE comparador[h] != comparador[v]
		ENTÃO quadrado <-- FALSO
	FIM
FIM
SE quadrado=VERDADEIRO
ENTÃO ESCREVA "É quadrado mágico"
SENÃO ESCREVA "Não é"
FIM_ALGORITMO
