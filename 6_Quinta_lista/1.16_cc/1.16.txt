ALGORITMO
DECLARE matrix[3,3],matriz[3,3],i,j,s,r NUMÉRICO
PARA i <-- 1 ATÉ 3 FAÇA
INÍCIO
	PARA j <-- 1 ATÉ 3 FAÇA
	INÍCIO
		LEIA matriz[i,j]
		SE i=1
		ENTÃO matrix[j, i+2] <-- matriz[i,j]
		SENÃO SE i=2
		ENTÃO matrix[j,i] <-- matriz[i,j]
		SENÃO SE i=3
		ENTÃO matrix[j,i-2] <-- matriz[i,j]
	FIM
FIM
PARA s <-- 1 ATÉ 3 FAÇA
INÍCIO
	PARA r <-- 1 ATÉ 3 FAÇA
	INÍCIO
		ESCREVA matrix[s,r], end=" "
	FIM
	ESCREVA "\n"
FIM
FIM_ALGORITMO
