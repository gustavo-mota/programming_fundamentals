ALGORITMO
DECLARE	ang,nfat,sen,cont NUMÉRICO
	soma=FALSO LÓGICO
ESCREVA "Digite um angulo"
LEIA ang
sen<==1
cont<==1
ang<==ang/(180/3.14)
FUNÇÃO fatorial(num)
	nfat<==1
	ENQUANTO (num>1) FAÇA
	INÍCIO
		nfat<==nfat*num
		num<==num-1
	FIM
	RETORNAR nfat
ENQUANTO (cont<=14) FAÇA
INÍCIO
	SE soma=VERDADEIRO
	ENTÃO	INÍCIO
		sen+=(ang**(cont))/fatorial(cont)
		soma=FALSO
		FIM
	SENÃO	INÍCIO
		sen-=(ang**(cont))/fatorial(cont)
		soma=VERDADEIRO
		FIM
	cont<==cont+1
FIM
ESCREVA sen

FIM_ALGORITMO
