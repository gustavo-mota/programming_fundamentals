qimpar<==0

INÍCIO
    	ESCREVA "Digite um inteiro negativo para descontinuar."
    	ESCREVA 'Digite um valor'

	ENTÃO	INÍCIO

        	SE (n MOD 2 =0)
		ENTÃO	INÍCIO

            		SE (n>parmaior)
                	ENTÃO	INÍCIO

                		qpar=qpar+1

				FIM

			FIM

        	SENÃO	INÍCIO

			SE (n>imparmaior)
			ENTÃO	INÍCIO
		                imparmaior=n

				FIM

	            	SENÃO 	INÍCIO
				qimpar<==qimpar+1

			FIM

		FIM

	SENÃO	INÍCIO
        	mediapar<==par/qpar

		FIM
FIM
