ALGORITMO
DECLARE s,numerador,n,soma1,soma2 NUMÉRICO
s<==0
ESCREVA "Digite um valor"
LEIA n
numerador<==1
soma1<==0
soma2<==0
denominador<==n
ENQUANTO (denominador>1) FAÇA
INÍCIO
	s<==numerador/denominador
	denominador<==denominador-1
	numerador<==numerador+1
	soma1<==soma1+s
FIM
ENQUANTO (denominador<n) FAÇA
INÍCIO
	s<==denominador/numerador
	denominador<==denominador+1
	numerador<==numerador-1
	soma2<==soma2+s
FIM
ESCREVA soma1+soma2

FIM_ALGORITMO
